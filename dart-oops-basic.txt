//   create is an constructor
    Dog(this.name, this.breed, this.gender, this.age, this.isFavorite);

  void bark()
  {
    print("Dog is barking");
  }
  
  void eat(int foodWeight)
  {
    
    if (foodWeight > 400) {
      print("Dog is eating more");
    } else {
      
      bark();
    }
     
    
  }

  }

    return isFavorite;
  }
  
  
}
